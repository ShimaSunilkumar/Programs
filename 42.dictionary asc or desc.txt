a = {'Apple': 1, 'Grapes': 3, 'Orange': 2, 'Watermelon': 4}
print("Dictionary : ",a)
print("ascending order using keys : ",sorted(a.keys()))
print("ascending order using values : ",sorted(a.values()))
print("descending order using keys : ",sorted(a.keys(),reverse=True))
print("descending order using values : ",sorted(a.values(),reverse=True))
